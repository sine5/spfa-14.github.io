#include <iostream>
using namespace std;
int a[1001],ans;
int main() {
    int n,avg=0;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
        avg+=a[i];
    }
    avg/=n;
    for(int i=0;i<n;i++){
        a[i]=a[i]-avg;
    }
    for(int i=0;i<n;i++){
        if (a[i]==0)continue;
        a[i+1]+=a[i];
        ans++;
    }
    cout<<ans<<endl;
    return 0;
}
