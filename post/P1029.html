#include<cstdio>
#include<algorithm>
using namespace std;
int gcd(int a,int b){
    if(b==0)return a;
    return gcd(b,a%b);
}
int lcm(int gcd,int a,int b){
    return a*b/gcd;
}
int x,y,sum;
int main(){
    scanf("%d%d",&x,&y);
    for(int i=x;i<=min(y,100000);i++){
        for(int j=i;j<=min(y,100000);j++){
            int g=gcd(i,j);
            int l=lcm(g,i,j);
            if(g==x&&l==y)sum++;
        }
    }
    printf("%d\n",sum*2);
    return 0;
}
