#include<bits/stdc++.h>
using namespace std;
int sum=0;
void f(int n,int k,int m){
    if(k==1){
        sum++;
        return;
    }
    for(int i=m;i<=n/k;i++){
        f(n-i,k-1,i);
    }
}
int main(){
    int n,k;
    cin>>n>>k;
    f(n,k,1);
    cout<<sum<<endl;
    return 0;
}
