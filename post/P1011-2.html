#include<cstdio>
#define maxn 21
using namespace std;
struct station{
  int ias;
  int izs;
  int oas;
  int ozs;
  int tas;
  int tzs;
}s[maxn];
int a,n,m,x;
int main(){
  scanf("%d%d%d%d",&a,&n,&m,&x);
  s[1].ias=1;
  s[1].izs=0;
  s[1].oas=0;
  s[1].ozs=0;
  s[1].tas=1;
  s[1].tzs=0;
  s[2].ias=0;
  s[2].izs=1;
  s[2].oas=0;
  s[2].ozs=1;
  s[2].tas=1;
  s[2].tzs=0;
  for(int i=3;i<=n;i++){
    s[i].ias=s[i-1].ias+s[i-2].ias;
    s[i].izs=s[i-1].izs+s[i-2].izs;
    s[i].oas=s[i-1].ias;
    s[i].ozs=s[i-1].izs;
    s[i].tas=s[i-1].tas+s[i].ias-s[i].oas;
    s[i].tzs=s[i-1].tzs+s[i].izs-s[i].ozs;
  }
  int tempa=s[n-1].tas;
  int tempz=s[n-1].tzs;
  int ts=m;
  ts-=tempa*a;
  int zval=ts/tempz;
  printf("%d\n",s[x].tas*a+s[x].tzs*zval);
  return 0;
}
