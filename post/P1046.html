#include<iostream>
using namespace std;
int main(){
    int apple[10],c=0,h;
    for(int i=0;i<10;i++){
        cin>>apple[i];
    }
    cin>>h;
    h+=30;
    for(int i=0;i<10;i++){
        if(apple[i]<=h)c++;
    }
    cout<<c;
    return 0;
}
