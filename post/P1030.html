#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
string as,bs;
void gp(string a,string b){
    if(a.length()<=0)return;
    char mid=b[b.length()-1];
    printf("%c",mid);
    if(a.length()<=1)return;
    int mn;
    for(int i=0;i<a.length();i++){
        if(a[i]==mid){
            mn=i;
            break;
        }
    }
    gp(a.substr(0,mn),b.substr(0,mn));
    gp(a.substr(mn+1),b.substr(mn,b.length()-mn-1));

}
int main(){
    string a,b;
    cin>>as>>bs;
    gp(as,bs);
    printf("\n");
    return 0;
}
