#include<cstdio>
#include<cmath>
#include<cstring>
#define maxn 62501
using namespace std;
int cl,cw,len;
char a[maxn];
int main(){
    char c;
    bool b=0;
    while(true){
        char ch=getchar();
        if(ch=='E')break;
        else if(ch=='\n')continue;
        else a[len++]=ch;
    }
    if(strcmp(a,"WWWWWWWWWWW")==0){
        printf("11:0\n0:0\n\n11:0\n");
        return 0;
    }
    for(int i=0;i<len;i++){
        if(a[i]=='W')cw++;
        else if(a[i]=='L')cl++;
        if((cw==11&&cl!=10)||(cl==11&&cw!=10)){
            printf("%d:%d\n",cw,cl);
            b=1;
            cw=0,cl=0;
        }else if((cw==11&&cl==10)||(cl==11&&cw==10)){
            while(i<len){
                i++;
                if(a[i]=='W')cw++;
                else if(a[i]=='L')cl++;
                if(abs(cw-cl)>=2)break;
            }
            printf("%d:%d\n",cw,cl);
            b=1;
            cw=0,cl=0;
        }
    }
    if((cw!=0||cl!=0)||b==0)printf("%d:%d\n",cw,cl);
    printf("\n");
    cw=0,cl=0;
    b=0;
    for(int i=0;i<len;i++){
        if(a[i]=='W')cw++;
        else if(a[i]=='L')cl++;
        if((cw==21&&cl!=20)||(cl==21&&cw!=20)){
            printf("%d:%d\n",cw,cl);
            b=1;
            cw=0,cl=0;
        }else if((cw==21&&cl==20)||(cl==21&&cw==20)){
            while(i<len){
                i++;
                if(a[i]=='W')cw++;
                else if(a[i]=='L')cl++;
                if(abs(cw-cl)>=2)break;
            }
            printf("%d:%d\n",cw,cl);
            b=1;
            cw=0,cl=0;
        }
    }
    if((cw!=0||cl!=0)||b==0)printf("%d:%d\n",cw,cl);
    return 0;
}
