#include<bits/stdc++.h>
using namespace std;
int main(){
    int l,n;
    int maxn=0,minn=0;
    cin>>l;
    cin>>n;
    for(int i=0;i<n;i++){
        int p;
        cin>>p;
        maxn=max(maxn,max(l-p+1,p));
        minn=max(minn,min(l-p+1,p));
    }
    cout<<minn<<" "<<maxn<<endl;

    return 0;
}
