#include<iostream>
#include<cstdio>
using namespace std;
int a[10001][5];
int main(){
    int al,b,n;
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i][1]>>a[i][2]>>a[i][3]>>a[i][4];
        a[i][3]+=a[i][1];
        a[i][4]+=a[i][2];
    }
    cin>>al>>b;
    for(int i=n;i>=1;i--){
        if(a[i][1]<=al&&a[i][2]<=b&&a[i][3]>=al&&a[i][4]>=b) {
            cout<<i<<endl;
            return 0;
        }
    }
    cout<<-1<<endl;
    return 0;
}
