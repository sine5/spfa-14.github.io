#include<bits/stdc++.h>
using namespace std;
int w[105],v[105];
int dp[1005];
//dp[j]=max(dp[j-w[i]]+v[i], dp[j]);
int main(){
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=m;i++){
        cin>>w[i]>>v[i];
    }
    for(int i=1;i<=m;i++){
        for(int j=n;j>=1;j--){
            if(j>=w[i]){
                dp[j]=max(dp[j-w[i]]+v[i], dp[j]);
            }
        }
    }
    cout<<dp[n]<<endl;
    return 0;
}
